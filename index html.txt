<script>
let members = [], payments = [];

function addMembers() {
  const input = document.getElementById('member-input');
  const lines = input.value.split('\n').map(l => l.trim()).filter(l => l.length > 0);
  lines.forEach(name => {
    if (!members.includes(name) && members.length < 20 && name.length <= 10) members.push(name);
  });
  input.value = '';
  renderChips(); renderPayments();
}

function removeMember(name) {
  members = members.filter(m => m !== name);
  renderChips(); renderPayments();
}

function renderChips() {
  const el = document.getElementById('chips');
  el.innerHTML = members.length === 0
    ? '<span style="font-size:13px;color:#aaa;">まだメンバーがいません</span>'
    : members.map(m => `<div class="chip">${m}<span class="chip-remove" onclick="removeMember('${m}')">×</span></div>`).join('');
}
